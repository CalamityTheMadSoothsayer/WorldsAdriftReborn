CREATE TABLE IF NOT EXISTS characterdetails (
    id SERIAL PRIMARY KEY,
    userKey VARCHAR(255) NOT NULL,
    characterUid VARCHAR(255),
    name VARCHAR(255),
    server VARCHAR(255),
    serverIdentifier VARCHAR(255),
    cosmetics JSONB,
    isMale BOOLEAN,
    seenIntro BOOLEAN,
    skippedTutorial BOOLEAN,
    universalColorsHairColorR FLOAT,
    universalColorsHairColorG FLOAT,
    universalColorsHairColorB FLOAT,
    universalColorsHairColorA FLOAT,
    universalColorsSkinColorR FLOAT,
    universalColorsSkinColorG FLOAT,
    universalColorsSkinColorB FLOAT,
    universalColorsSkinColorA FLOAT,
    universalColorsLipColorR FLOAT,
    universalColorsLipColorG FLOAT,
    universalColorsLipColorB FLOAT,
    universalColorsLipColorA FLOAT
);
